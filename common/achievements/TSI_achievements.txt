german_century = {
	possible = {
	}
	happened = {
		exists = c:GER
		this = c:GER
		or = {
			NOT = { 
				c:GBR = {
					is_country_alive = yes
				}
			}
			c:GBR = { 
				country_rank < rank_value:major_power 
			}
		}
		or = {
			NOT = { 
				c:OCC = {
					is_country_alive = yes
				}
			}
			c:OCC = { 
				country_rank < rank_value:major_power 
			}
		}
		or = {
			NOT = { 
				c:UKR = {
					is_country_alive = yes
				}
			}
			c:UKR = { 
				country_rank < rank_value:major_power 
			}
		}
	}
}
